\section*{Testing Switch Environment}

Input "Apple":
\begin{SwitchEnv}{Apple}
    \Case{Apple}{This is an apple in an environment.}
    \Case{Orange}{This is an orange in an environment.}
    \Default{Fruit not found!}
\end{SwitchEnv}

\medskip

Input "Grape":
\begin{SwitchEnv}{Grape}
    \Case{Apple}{This is an apple in an environment.}
    \Case{Orange}{This is an orange in an environment.}
    \Default{Fruit not found!}
\end{SwitchEnv}

\def\fruit{Apple}
\begin{SwitchEnv}{\fruit}
